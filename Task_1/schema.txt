Departments(_name, abbreviation)

Programmes(_name, abbreviation)

Courses(_code,name,credits)

HasPrerequisites(_course, _prereq_course)
	course -> Courses.code
	prereq_course -> Courses.code

CourseType(_type)	

TypeOf(_course, _coursetype)
	course -> Courses.code
	coursetype -> CourseTypes.type	

LimitedCourses(_code, name, credits, #students)	

HasProgrammes(_department, _programme)
	department -> Departments.name
	programme -> Programmes.name

HasBranches(_programme, _branch)
	programme -> Programmes.name
	branch -> Branches.name

HasMandatory(_programme, _branch, _course)
	programme -> Programmes.name
	branch -> Branches.name
	course -> Courses.code
	
GivenBy(_department, _course)
	department -> Departments.name
	course -> Courses.code

HasRecommended(_programme, _branch, _course)
	programme -> Programmes.name
	branch -> Branches.name
	course -> Courses.code

Users(_id, name) 
	
Students(_id, name, branch, programme)
	branch -> Branches.name
	programme -> Programmes.name

RegisteredTo(_student, _course)
	student -> Students.id
	course -> Courses.code
	
Finished(_student, _course, grade)
	student -> Students.id
	course -> Courses.code

InQueue(_student, _limitedCourse, timeRegistered)
	student -> Students.id
	limitedCourse -> LimitedCourses.code

Administrators(_id, name, role)

HasAccess(_id, accessCourse)
	id -> Administrators.id
	accessCourse -> Courses.code
